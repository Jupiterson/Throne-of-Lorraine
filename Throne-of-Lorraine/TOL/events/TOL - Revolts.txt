#Finnish Revolt 
country_event = {
	id = 998500
	title = "Finland in revolt!"
	desc = "The Finnishs are revolting against our rule!"
	picture = "crete_revolt"
#	fire_only_once = yes

	trigger = {
		any_pop = {
			has_pop_culture = finnish 
			militancy = 6 
			location = { is_core = FIN }
		}
		tag = SWE
		343 = { owned_by = THIS }
		#war = no
		NOT = { exists = FIN }
		NOT = { has_country_modifier = revolt_cooldown }
	}

	mean_time_to_happen = {
		months = 48

		modifier = {
			factor = 0.5
			average_militancy = 6
		}

		modifier = {
			factor = 0.5
			has_recently_lost_war = yes
		}
	}

	option = {
		name = "Crush them!"
		any_owned = {
			limit = { 
			is_core = FIN
            NOT = { is_core = SAP }
			}
			secede_province = FIN 
		}
		diplomatic_influence = {
			who = FIN 
			value = -400
		}
		casus_belli = {
			target = FIN 
			type = conquest
			months = 12
		}
		add_country_modifier = { name = revolt_cooldown duration = 1825 }
		war = {
			target = FIN 
			attacker_goal = { casus_belli = conquest }
			defender_goal = { casus_belli = status_quo }
		}
		ai_chance = {
			factor = 100
		}
	}
	
}
