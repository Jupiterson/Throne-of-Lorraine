country_event = {
	id = 310001
	title = "FEU1.T"
	desc = "FEU1.D"
	picture = "Guangxi_revolt_pic"
	
	trigger = {
		tag = QNG
		year = 1839
		NOT = { has_country_flag = victory_of_feudatories }
		NOT = { has_country_flag = victory_of_Buyeo }		
		NOT = { has_country_flag = Guangdong_bye_bye }
		NOT = { war_with = GXI }
	}
	mean_time_to_happen = {
		months = 10
		modifier = {
			factor = 0.5
			year = 1840
		}
		modifier = {
			factor = 0.5
			year = 1841
		}
		modifier = {
			factor = 0.5
			year = 1842
		}
		modifier = {
			factor = 0.25
			year = 1843
		}
	}	
	option = {
		name = "We are powerless..."
		prestige = -10
		money = 50000
		relation = { who = GXI value = -100 }
		any_owned = {
			limit = {
				or = {
					region = CHI_1499
					region = CHI_1493
				}
			}
			secede_province = GXI
		}
		CHI_1499 = { add_core = GXI }
		CHI_1493 = { add_core = GXI }
		GXI = { country_event = 310002 }
		any_pop = {
			limit = { is_primary_culture = no }
			militancy = 1
		}
		set_country_flag = Guangdong_bye_bye
	}
}
